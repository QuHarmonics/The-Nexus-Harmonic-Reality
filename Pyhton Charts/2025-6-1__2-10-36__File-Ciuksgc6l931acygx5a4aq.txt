import matplotlib.pyplot as plt

# Original Byte 1 (User)
user_byte1 = [1, 4, 1, 5, 9, 2, 6, 5]

# Grok's Byte 1
grok_byte1 = [3, 1, 2, 5, 6, 4, 5, 4]

# Triadic chunks (every 4)
user_triadic_chunks = [user_byte1[i:i+4] for i in range(0, len(user_byte1), 4)]
grok_triadic_chunks = [grok_byte1[i:i+4] for i in range(0, len(grok_byte1), 4)]

# Plotting
fig, axs = plt.subplots(2, 1, figsize=(10, 6), sharex=True)

# Plot User Triadic Chunks
for i, chunk in enumerate(user_triadic_chunks):
    axs[0].plot(range(i * 4, i * 4 + len(chunk)), chunk, marker='o', label=f'Triad {i+1}')
axs[0].set_title('User Byte 1 - Triadic Chunks')
axs[0].legend()
axs[0].grid(True)

# Plot Grok Triadic Chunks
for i, chunk in enumerate(grok_triadic_chunks):
    axs[1].plot(range(i * 4, i * 4 + len(chunk)), chunk, marker='x', label=f'Triad {i+1}')
axs[1].set_title("Grok's Byte 1 - Triadic Chunks")
axs[1].legend()
axs[1].grid(True)

plt.tight_layout()
plt.show()
