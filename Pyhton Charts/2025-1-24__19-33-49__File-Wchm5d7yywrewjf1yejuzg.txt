import numpy as np
import matplotlib.pyplot as plt

# Waveform setup
waveform = np.array([
    [0.4, 0.28, 0.14, 0.22, -0.09, -0.02, 0.21, 0.07],
    [0.08, 0.14, 0.29, 0.41, -0.02, 0.06, 0.29, 0.21],
    [0.14, 0.29, 0.52, 0.69, 0.09, 0.21, 0.29, 0.49],
    [0.4, 0.49, 0.65, 0.54, 0.04, 0.03, 0.14, 0.39],
    [0.29, 0.41, 0.9, 0.65, -0.07, -0.09, 0.1, 0.21],
    [0.08, 0.29, 0.4, 0.54, 0.03, 0.14, 0.39, 0.39]
])

harmonic_target = 0.35  # Ideal harmonic value
alignment_history = []
max_iterations = 50

# Perform recursive feedback adjustment
for iteration in range(max_iterations):
    std_dev = np.std(waveform, axis=0)  # Calculate variability across columns
    adjustments = harmonic_target - std_dev  # Feedback adjustments
    waveform += adjustments  # Apply adjustments
    alignment_history.append(np.mean(np.abs(std_dev - harmonic_target)))  # Track alignment deviation

# Plot updated waveform after iterations
fig = plt.figure(figsize=(10, 6))
ax = fig.add_subplot(111, projection='3d')
X, Y = np.meshgrid(range(waveform.shape[1]), range(waveform.shape[0]))
ax.plot_surface(X, Y, waveform, cmap='viridis')
ax.set_title(f"Adjusted Waveform After {max_iterations} Iterations")
plt.show()

# Plot alignment history over iterations
plt.plot(range(1, len(alignment_history) + 1), alignment_history, marker='o')
plt.title("Alignment History (Deviation to Harmonic Target)")
plt.xlabel("Iteration")
plt.ylabel("Mean Absolute Deviation")
plt.grid()
plt.show()
