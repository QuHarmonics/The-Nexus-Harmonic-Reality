import numpy as np
import matplotlib.pyplot as plt

# Time axis
t = np.linspace(0, 2 * np.pi, 1000)

# Odd integer waveforms as sine waves with frequencies 3 and 5
wave3 = np.sin(3 * t)
wave5 = np.sin(5 * t)

# Harmonic midpoint (average of the two waves) representing '4'
harmonic_midpoint = (wave3 + wave5) / 2

# Plotting
plt.figure(figsize=(10, 6))
plt.plot(t, wave3, label='Odd 3: sin(3t)', linewidth=1)
plt.plot(t, wave5, label='Odd 5: sin(5t)', linewidth=1)
plt.plot(t, harmonic_midpoint, label='Even 4 (avg): (sin(3t)+sin(5t))/2', linewidth=2, color='black')
plt.title('Wave Interference: 3 and 5 Folding into 4')
plt.xlabel('t')
plt.ylabel('Amplitude')
plt.legend()
plt.grid(True)
plt.tight_layout()
plt.show()
