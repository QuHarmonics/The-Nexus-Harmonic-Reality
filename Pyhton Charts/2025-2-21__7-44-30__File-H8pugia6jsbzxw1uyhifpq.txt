# Adjusted plotting: Each triangle should be placed sequentially without overlap
# C = 0 should create a vertical line

# Define the given extended π digits sequence
extended_pi_digits = [1, 4, 1, 5, 9, 2, 6, 5, 3, 5, 1, 8, 9, 0, 7, 9, 1, 3, 2, 1, 3, 8, 2, 4, 6, 0, 2, 6, 1, 4, 3, 1]

# Extract (B, A, C) triplets using three sequential values at a time
triangles_bac = [(extended_pi_digits[i], extended_pi_digits[i+1], extended_pi_digits[i+2])
                 for i in range(0, len(extended_pi_digits) - 2, 3)]

# Set up sequential plotting along the x-axis
x_offset = 0

plt.figure(figsize=(14, 6))

for i, (b, a, c) in enumerate(triangles_bac):
    # Triangle vertices based on B (right side), A (left side), and C (bottom width)
    A = (x_offset, 0)
    B = (x_offset + c, 0)  # Base width along the x-axis

    if c == 0:
        # If C = 0, create a vertical line
        Cx, Cy = x_offset, a  # Vertical line with height A
    else:
        Cx = x_offset + (c / 2)  # Place peak at midpoint of base
        Cy = a  # Height using A as the vertical side

    C = (Cx, Cy)

    # Plot the triangle (or vertical line if C = 0)
    if c == 0:
        plt.plot([A[0], C[0]], [A[1], C[1]], marker='o', linestyle='-', color='black', label=f"Sector End {i+1}" if i == 0 else "")
    else:
        x_coords = [A[0], B[0], C[0], A[0]]
        y_coords = [A[1], B[1], C[1], A[1]]
        plt.plot(x_coords, y_coords, marker='o', linestyle='-', label=f"Triangle {i+1}" if i == 0 else "")

    # Annotate points
    plt.text(A[0], A[1], "A", fontsize=10, verticalalignment='bottom', horizontalalignment='right')
    plt.text(B[0], B[1], "B", fontsize=10, verticalalignment='bottom', horizontalalignment='left')
    plt.text(C[0], C[1], "C", fontsize=10, verticalalignment='bottom', horizontalalignment='left')

    # Move x-offset for the next triangle, ensuring they touch but do not overlap
    x_offset += c if c != 0 else 1  # Move by base width or 1 if it's a vertical line

# Formatting the plot
plt.title("Sequential π Triangle Representation (B, A, C) with C as Base Width and Sector Markers")
plt.xlabel("X-Axis (Sequential Placement)")
plt.ylabel("Y-Axis (Triangle Height)")
plt.axhline(0, color='black', linewidth=0.5)
plt.grid(True)
plt.legend()
plt.show()
