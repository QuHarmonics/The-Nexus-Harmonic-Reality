import matplotlib.pyplot as plt
import numpy as np
import pandas as pd

# Atomic numbers and corresponding element counts by binary group
binary_groups = [2, 4, 8, 16, 32, 64]
start_atomic = 1

# Create DataFrame to hold element ranges
group_data = []
current_atomic = start_atomic
for size in binary_groups:
    end_atomic = current_atomic + size - 1
    group_data.append({
        'Group Size': size,
        'Start Atomic Number': current_atomic,
        'End Atomic Number': end_atomic
    })
    current_atomic = end_atomic + 1

df_binary_ranges = pd.DataFrame(group_data)

# Plot the grouping visually
plt.figure(figsize=(10, 6))
x = [f'{row["Start Atomic Number"]}-{row["End Atomic Number"]}' for _, row in df_binary_ranges.iterrows()]
y = df_binary_ranges['Group Size']
plt.bar(x, y, color='skyblue', edgecolor='black')
plt.title("Recursive Harmonic Grouping in the Periodic Table")
plt.xlabel("Atomic Number Range")
plt.ylabel("Group Size (Elements)")
plt.grid(True)
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()
