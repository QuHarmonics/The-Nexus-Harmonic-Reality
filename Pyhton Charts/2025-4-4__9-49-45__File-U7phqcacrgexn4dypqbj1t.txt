# Spiral plot: SHA1 and SHA2 as true ripples from center
# Angle increases linearly, radius follows data value (spiral wave)

def spiral_wave_plot(data, title, color):
    n = len(data)
    theta = np.linspace(0, 4 * np.pi, n)  # 2 full revolutions (spiral)
    r = data.astype(float)
    x = r * np.cos(theta)
    y = r * np.sin(theta)

    plt.figure(figsize=(6, 6))
    plt.plot(x, y, color=color)
    plt.fill(x, y, color=color, alpha=0.25)
    plt.title(title)
    plt.axis('equal')
    plt.axis('off')
    plt.tight_layout()
    plt.show()
