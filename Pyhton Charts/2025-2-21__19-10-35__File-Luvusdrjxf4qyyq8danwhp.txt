# Function to count digit frequencies
def count_frequencies(data, base=16):
    counter = Counter(data)
    sorted_keys = sorted(counter.keys(), key=lambda x: int(x, base))
    return {k: counter[k] for k in sorted_keys}

# Convert all to numeric form
pi_decimal_freq = count_frequencies(pi_decimal, base=10)
