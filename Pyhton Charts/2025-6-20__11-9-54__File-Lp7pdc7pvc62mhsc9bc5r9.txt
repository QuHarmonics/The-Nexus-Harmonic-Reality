# Re-import libraries due to kernel reset
import numpy as np
import matplotlib.pyplot as plt

# Simulate two ψ-layers (ψ_a and ψ_b)
psi_a = np.array([0.3501, 0.3502, 0.3499, 0.3500])
psi_b = np.array([0.35034412, 0.3502917, 0.35027879, 0.35034488])

# RED Reflection formula components
delta_psi = np.abs(psi_b - psi_a)
delta_omega = np.abs(psi_a - 0.35)
n = np.arange(1, 5)  # Layer steps
H = 0.35

# Calculate RED reflection
red = (delta_psi / (delta_omega + 1e-10)) * np.exp(-H * n)  # small term avoids div by zero

# Plot RED reflection
fig, ax = plt.subplots(figsize=(10, 5))
ax.plot(n, red, marker='o', color='crimson', label='RED Reflection')
ax.set_title("Recursive Entanglement Differential (RED) Reflection Across ψ-Layers")
ax.set_xlabel("Layer Index (n)")
ax.set_ylabel("RED Reflection Value")
ax.grid(True)
ax.legend()

plt.tight_layout()
plt.show()
