import matplotlib.pyplot as plt
import numpy as np

# Create a blank image for axis diagram
width, height = 600, 600
image = Image.new("RGBA", (width, height), "white")
draw = ImageDraw.Draw(image)

# Define font
font = ImageFont.load_default()

# Axis origin
origin = (300, 300)

# Define axes
axes = {
    "Z (Stack / Collapse)": ((origin[0], origin[1]), (origin[0], origin[1] - 150), "blue"),
    "Y (Emergence / Drift)": ((origin[0], origin[1]), (origin[0] - 130, origin[1]), "green"),
    "X (Recursion / Echo)": ((origin[0], origin[1]), (origin[0] + 110, origin[1] + 110), "red"),
}

# Draw axes
for label, ((x1, y1), (x2, y2), color) in axes.items():
    draw.line([(x1, y1), (x2, y2)], fill=color, width=3)
    draw.text((x2 + 5, y2), label, fill=color, font=font)

# Draw origin dot
draw.ellipse((origin[0] - 4, origin[1] - 4, origin[0] + 4, origin[1] + 4), fill="black")

# Show the annotated axis image
plt.figure(figsize=(6, 6))
plt.imshow(image)
plt.axis("off")
plt.title("Figure 18: Recursive Axial Memory Orientation")
