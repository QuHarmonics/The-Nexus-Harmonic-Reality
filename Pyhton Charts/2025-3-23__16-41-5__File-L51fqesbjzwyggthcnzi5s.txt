import matplotlib.pyplot as plt

# First 64 digits of Pi after the decimal (excluding the "3.")
pi_digits_str = "14159265358979323846264338327950288419716939937510" \
                "58209749445923078164062862089986280348253421170679"
pi_digits = [int(d) for d in pi_digits_str[:64]]

# Calculate cumulative odd and even counts
odd_counts = []
even_counts = []
odd_total = 0
even_total = 0

for d in pi_digits:
    if d % 2 == 0:
        even_total += 1
    else:
        odd_total += 1
    odd_counts.append(odd_total)
    even_counts.append(even_total)

# Plotting
plt.figure(figsize=(12, 6))
plt.plot(range(1, 65), odd_counts, label="Odd Count", linewidth=2)
plt.plot(range(1, 65), even_counts, label="Even Count", linewidth=2)
plt.title("Odd vs Even Count in First 64 Digits of Pi")
plt.xlabel("Digit Position")
plt.ylabel("Cumulative Count")
plt.legend()
plt.grid(True)
plt.show()
