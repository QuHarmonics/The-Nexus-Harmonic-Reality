import numpy as np
import matplotlib.pyplot as plt
from hashlib import sha256

# Recalculate original byte data (real block header from block #722460)
header_hex = (
    "040000006fe28c0ab6f1b372c1a6a246ae63f74f931e8365e15a089c68d61900"
    "00000000982051fd1e4ba744bbbe680e1fee14677ba1a3c3540bf7b1cdb606e8"
    "57233e0e61bc6649ffff001d01e36299"
)
header_bytes = bytes.fromhex(header_hex)
sha1_bytes = sha256(header_bytes).digest()
sha2_bytes = sha256(sha1_bytes).digest()

# Convert to arrays
sha1_array = np.array([b for b in sha1_bytes], dtype=np.float64)
sha2_array = np.array([b for b in sha2_bytes], dtype=np.float64)

# Define overlay spiral plotting function
def spiral_overlay_plot(data1, data2, label1, label2, color1, color2, title):
    n = len(data1)
    theta = np.linspace(0, 4 * np.pi, n)
    
    r1 = data1.astype(float)
    r2 = data2.astype(float)
    
    x1 = r1 * np.cos(theta)
    y1 = r1 * np.sin(theta)
    
    x2 = r2 * np.cos(theta)
    y2 = r2 * np.sin(theta)
    
    plt.figure(figsize=(7, 7))
    plt.plot(x1, y1, label=label1, color=color1)
    plt.plot(x2, y2, label=label2, color=color2, linestyle='--', alpha=0.7)
    plt.fill(x1, y1, color=color1, alpha=0.2)
    plt.fill(x2, y2, color=color2, alpha=0.2)
    plt.title(title)
    plt.axis('equal')
    plt.axis('off')
    plt.legend()
    plt.tight_layout()
    plt.show()

# Execute overlay spiral plot
spiral_overlay_plot(
    sha1_array, sha2_array,
    label1="SHA1: Memory Echo",
    label2="SHA2: Memory Collapse",
    color1='orange', color2='red',

)
