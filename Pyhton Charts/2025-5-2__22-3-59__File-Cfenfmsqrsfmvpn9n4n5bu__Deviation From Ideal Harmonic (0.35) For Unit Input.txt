import numpy as np
import matplotlib.pyplot as plt

# Define axis strengths and labels
axis_labels = ["Magnetic", "Strong Nuclear", "Weak Nuclear"]
axis_strengths = np.array([0.30, 0.40, 0.25])

# Assume a unit input (1.0 energy) into each axis individually and compute the output
# Output is deviation from ideal (0.35) harmonic value when that axis acts alone

ideal_harmonic = 0.35
deviation = np.abs(axis_strengths - ideal_harmonic)

# Plotting the deviation from ideal harmonic output
plt.figure(figsize=(8, 5))
bars = plt.bar(axis_labels, deviation, color=["steelblue", "firebrick", "darkgreen"])
plt.axhline(0, color="gray", linestyle="--", linewidth=1)
plt.title("Deviation from Ideal Harmonic (0.35) for Unit Input")
plt.ylabel("Absolute Deviation from 0.35")
plt.xlabel("Axis (Unit Input)")

# Annotate bars
for bar in bars:
    yval = bar.get_height()
    plt.text(bar.get_x() + bar.get_width()/2.0, yval + 0.005, f"{yval:.3f}", ha='center', va='bottom')

plt.tight_layout()
plt.grid(axis='y', linestyle=':', alpha=0.7)
plt.show()
