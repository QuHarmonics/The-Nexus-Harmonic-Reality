# â”€â”€ Critical radius per Nexus3 collapse rule â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
def critical_radius(m1, m2):
    # phase-lock radius where delta-resonance flips
    return np.sqrt(G * m1 * m2 / (hbar * c))

# â”€â”€ Loopback-gravity suppression curve â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
def suppression_ratio(r, m1, m2):
    # G_loop/G_Newton = exp[-(r_c/r)^2]
    rc = critical_radius(m1, m2)
    return np.exp(- (rc / r)**2)
