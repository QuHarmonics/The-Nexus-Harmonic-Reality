import numpy as np
import matplotlib.pyplot as plt

# Simulated window (e.g., gp120 region from ~7000–7500 bp)
t = np.linspace(0, 50, 300)
H = 0.35  # harmonic constant

# Simulated forces and energies
F_mis = 4.8
F_cor = 2.2
R0_mis = -5.7
R0_cor = 1.3

# Recursive harmonic fold function
def R(t, R0, F):
    return R0 * np.log(np.exp(H * F * t) + 1)

R_mis = R(t, R0_mis, F_mis)
R_cor = R(t, R0_cor, F_cor)
delta_R = R_mis - R_cor
Q_score = 30 + 18 * (1 - np.exp(-0.06 * t))

# Plot ΔR(t) and Q_score
fig, ax = plt.subplots(figsize=(12, 6))
ax.plot(t, delta_R, label='ΔR(t): Fold Discrepancy')
ax.plot(t, Q_score, label='Q_corrected(t): Peptide Harmony Gain')
ax.axhline(0, color='gray', linestyle='--', linewidth=0.5)
ax.set_title("gp120 Harmonic Correction via PSREQ Peptide")
ax.set_xlabel("Time (t)")
ax.set_ylabel("Energy Alignment")
ax.legend()
ax.grid(True)

plt.tight_layout()
plt.show()
