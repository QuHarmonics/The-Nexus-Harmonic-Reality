# Recursive step to compute second-level ratios (Level 2)
def compute_next_level(ratios):
    keys = list(ratios.keys())
    values = list(ratios.values())
    next_ratios = {}

            label = f"{keys[i]} / {keys[j]}"

    return next_ratios
