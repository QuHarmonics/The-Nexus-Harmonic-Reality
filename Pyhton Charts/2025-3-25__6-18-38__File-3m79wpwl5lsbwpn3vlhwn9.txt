import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

# Define mass values in kg, spanning small particles to large astronomical bodies
mass_values = np.logspace(-30, 5, num=20)  # Logarithmic scale from 10^-30 kg to 10^5 kg

# Calculate energy using Einstein's equation: E = mc^2
c = 3.0e8  # Speed of light in m/s
einstein_energy = mass_values * c**2

# Calculate energy using Recursive Gravity Loopback (Nexus2)
# Using a simplified recursive model: E_recursive = mc^2 * (1 + 0.35 * log(m))
nexus2_energy = mass_values * c**2 * (1 + 0.35 * np.log(mass_values + 1e-30))

# Calculate energy using MOND approximation
# MOND modifies Newtonian gravity at low accelerations, affecting gravitational dynamics
mond_acceleration_threshold = 1.2e-10  # m/s²
mond_energy = mass_values * c**2 * (1 + (mass_values / mond_acceleration_threshold) ** -0.5)

# Store results in a DataFrame
data = {
    "Mass (kg)": mass_values,
    "Einstein Energy (J)": einstein_energy,
    "Nexus2 Recursive Energy (J)": nexus2_energy,
    "MOND Energy (J)": mond_energy,
}

df_results = pd.DataFrame(data)

# Plot the results

plt.loglog(mass_values, nexus2_energy, 'r-', label="Recursive Gravity Loopback (Nexus2)")
plt.loglog(mass_values, mond_energy, 'g-.', label="MOND Energy Approximation")

plt.xlabel("Mass (kg)")
plt.ylabel("Energy (J)")
plt.title("Comparison of Einstein's Gravity, Recursive Gravity (Nexus2), and MOND")
