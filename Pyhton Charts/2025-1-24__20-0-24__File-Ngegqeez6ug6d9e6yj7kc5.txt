import hashlib
import numpy as np
import matplotlib.pyplot as plt

# Define two sample inputs and compute their SHA-256 hashes
input1 = "Hello, World!"
input2 = "Hello, Quantum!"

hash1 = hashlib.sha256(input1.encode()).hexdigest()
hash2 = hashlib.sha256(input2.encode()).hexdigest()

# Convert hashes into numeric arrays for comparison
def hash_to_numeric_array(hash_value):
    return np.array([int(hash_value[i:i+2], 16) for i in range(0, len(hash_value), 2)])

array1 = hash_to_numeric_array(hash1)
array2 = hash_to_numeric_array(hash2)

# Compute differences and visualize
difference = array1 - array2

# Visualization
plt.figure(figsize=(10, 6))

plt.subplot(3, 1, 1)
plt.plot(array1, label="Hash 1 (Numeric Representation)", color='blue')
plt.legend()

plt.subplot(3, 1, 2)
plt.plot(array2, label="Hash 2 (Numeric Representation)", color='green')
plt.legend()

plt.subplot(3, 1, 3)
plt.plot(difference, label="Difference (Hash 1 - Hash 2)", color='red')
plt.axhline(0, color='black', linestyle='--', linewidth=0.7)
plt.legend()

plt.tight_layout()
plt.show()
