import zipfile
import os
from PIL import Image
import matplotlib.pyplot as plt

# Define the path to the zip file and the extraction path
zip_path = "/mnt/data/Charts.zip"
extract_path = "/mnt/data/Charts"

# Extract the contents of the zip file
with zipfile.ZipFile(zip_path, 'r') as zip_ref:
    zip_ref.extractall(extract_path)

# Get list of image files in the extracted directory
image_files = sorted([
    os.path.join(extract_path, file)
    for file in os.listdir(extract_path)
    if file.lower().endswith(('.png', '.jpg', '.jpeg'))
])

# Load and display the next batch of 15 images
batch_size = 15
start_index = 15  # Adjust this to continue from the last batch
end_index = start_index + batch_size
selected_images = image_files[start_index:end_index]

# Plot the images
fig, axes = plt.subplots(5, 3, figsize=(15, 20))
axes = axes.flatten()

for ax, img_path in zip(axes, selected_images):
    img = Image.open(img_path)
    ax.imshow(img)
    ax.set_title(os.path.basename(img_path))
    ax.axis('off')

# Hide any unused subplots
for ax in axes[len(selected_images):]:
    ax.axis('off')

plt.tight_layout()
plt.show()
