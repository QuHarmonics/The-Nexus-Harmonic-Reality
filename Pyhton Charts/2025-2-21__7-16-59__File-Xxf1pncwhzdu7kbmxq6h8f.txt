# Plot π digits vs. their bit-length differences to visualize relationships

# Extract digits of π and corresponding bit-lengths
pi_digits_subset = [1, 4, 1, 5, 9, 2, 6, 5]
bit_lengths_subset = [bin(digit).count('1') for digit in pi_digits_subset]

# Compute bit-length differences
bit_differences_subset = [abs(bit_lengths_subset[i] - bit_lengths_subset[i - 1]) for i in range(1, len(bit_lengths_subset))]

# Generate x-axis values
x_values = list(range(1, len(pi_digits_subset) + 1))

# Plot π digits and bit-length differences
plt.figure(figsize=(8, 6))
plt.plot(x_values, pi_digits_subset, marker='o', linestyle='-', label="π Digits", color='blue')
plt.plot(x_values[:-1], bit_differences_subset, marker='s', linestyle='--', label="Bit-Length Differences", color='red')

# Formatting the plot
plt.title("π Digits vs. Bit-Length Differences")
plt.xlabel("Index in Sequence")
plt.ylabel("Value")
plt.axhline(0, color='black', linewidth=0.5)
plt.legend()
plt.grid(True)
plt.show()
