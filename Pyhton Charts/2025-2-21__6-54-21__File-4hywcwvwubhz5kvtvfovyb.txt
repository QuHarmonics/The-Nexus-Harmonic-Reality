import matplotlib.pyplot as plt
import numpy as np

# Given π's first 39 digits
pi_digits = "3141592653589793238462643383279502884197"
pi_values = [int(digit) for digit in pi_digits]  # Convert digits to integers

# Number of points to plot
num_points = len(pi_values)

# Generate (x, y) coordinates for each digit using a right-triangle expansion pattern
x_coords = [0]
y_coords = [0]

for i in range(1, num_points):
    # Calculate the next position based on the Pythagorean triangle movement
    a = pi_values[i-1]
    b = pi_values[i]
    c = np.sqrt(a**2 + b**2)  # Hypotenuse calculation
    
    # Append the new coordinates (x moves forward, y moves along the triangle)
    x_coords.append(x_coords[-1] + c / 2)  # Move x by half of the hypotenuse
    y_coords.append(y_coords[-1] + c / 2)  # Move y by the other half

# Plot the movement pattern
plt.figure(figsize=(10, 6))
plt.plot(x_coords, y_coords, marker='o', linestyle='-', color='blue', markersize=6, alpha=0.75)

# Annotate points with π's digits
for i in range(num_points):
    plt.text(x_coords[i], y_coords[i], str(pi_values[i]), fontsize=12, ha='right', va='bottom')

# Formatting the plot
plt.title("BBP Triangular Motion Through π’s Digits")
plt.xlabel("X-Coordinate (Sequential Hypotenuse Midpoints)")
plt.ylabel("Y-Coordinate (Pythagorean Expansion)")
plt.grid(True)
plt.show()
