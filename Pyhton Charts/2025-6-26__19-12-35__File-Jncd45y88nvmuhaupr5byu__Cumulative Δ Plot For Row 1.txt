import matplotlib.pyplot as plt

# Only plot Row 1 as requested
row_1 = [1, 6, 7, 7, 8, 9, 9, 10]
x_vals = [i + 1 for i in range(8)]  # Time steps starting from 1

# Plotting Row 1
plt.figure(figsize=(10, 6))
plt.plot(x_vals, row_1, marker='o', linestyle='-', linewidth=2, color='blue', label="Row 1")
plt.title("Cumulative Δ Plot for Row 1")
plt.xlabel("Time Step")
plt.ylabel("Cumulative Δ State")
plt.grid(True)
plt.legend()
plt.tight_layout()
plt.show()
