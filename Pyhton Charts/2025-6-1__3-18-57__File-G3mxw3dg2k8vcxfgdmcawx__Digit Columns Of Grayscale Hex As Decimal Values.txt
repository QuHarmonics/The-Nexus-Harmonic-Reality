import matplotlib.pyplot as plt
import pandas as pd

# Create a DataFrame based on the grayscale hex values converted earlier
# Sample: 0x111111 to 0x999999 step by 0x111111
hex_values = ['0x111111', '0x222222', '0x333333', '0x444444', '0x555555', '0x666666', '0x777777', '0x888888', '0x999999']
decimal_values = [int(h, 16) for h in hex_values]

# Break down each decimal into individual digits
digits = [[int(d) for d in str(d_val).zfill(8)] for d_val in decimal_values]

# Convert to DataFrame
df_digits = pd.DataFrame(digits, columns=[f"Digit_{i+1}" for i in range(8)])

# Plot each digit column
plt.figure(figsize=(12, 6))
for i in range(8):
    plt.plot(df_digits.index, df_digits[f"Digit_{i+1}"], marker='o', label=f'Digit {i+1}')

plt.title("Digit Columns of Grayscale Hex as Decimal Values")
plt.xlabel("Row Index")
plt.ylabel("Digit Value")
plt.legend()
plt.grid(True)
plt.tight_layout()
plt.show()
