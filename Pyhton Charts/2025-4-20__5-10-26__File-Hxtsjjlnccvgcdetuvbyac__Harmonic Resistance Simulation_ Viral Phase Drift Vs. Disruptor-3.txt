t = np.linspace(0, 10, 500)

# Viral phase drift simulation (resistance evolution over time)
# Simulate phase shift attempts by virus to escape resonance collapse
phase_shifts = [0, np.pi/6, np.pi/4, np.pi/3]  # increasing resistance phases
stability_base = np.exp(-0.3 * t)

# Generate resistant viral fold curves

    stability_base * np.cos(2 * np.pi * t + phase) for phase in phase_shifts
]

    plt.plot(t, curve, label=label, linestyle='--')
