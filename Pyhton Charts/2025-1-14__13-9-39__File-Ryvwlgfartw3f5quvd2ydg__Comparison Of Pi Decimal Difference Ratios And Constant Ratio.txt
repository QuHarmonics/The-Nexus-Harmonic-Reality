import matplotlib.pyplot as plt
import numpy as np

# Calculate the difference ratios for the first 300 decimal digits of π
from mpmath import mp

mp.dps = 310  # Set precision
pi_digits = [int(d) for d in str(mp.pi)[2:302]]  # Extract first 300 decimal digits of π

# Calculate differences
differences = [pi_digits[i] - pi_digits[i - 1] for i in range(1, len(pi_digits))]
ratios = [abs(differences[i] / (i + 1)) for i in range(len(differences))]

# Generate the constant ratio line
constant_ratio = [0.1111 for _ in range(len(ratios))]

# Plotting the results
plt.figure(figsize=(12, 6))
plt.plot(ratios, label="Ratios of Pi Differences", color="blue", alpha=0.7)
plt.plot(constant_ratio, label="Constant Ratio (0.1111)", color="orange", linestyle="--")
plt.title("Comparison of Pi Decimal Difference Ratios and Constant Ratio")
plt.xlabel("Index")
plt.ylabel("Ratio")
plt.legend()
plt.grid()
plt.show()
