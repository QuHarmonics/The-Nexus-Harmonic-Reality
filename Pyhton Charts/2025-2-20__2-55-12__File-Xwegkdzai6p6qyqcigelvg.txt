def fractal_phi_extraction(n, depth=10):
    """
    Extracts digits of φ using a fractal recursive approach inspired by KRRB.
    The method approximates non-sequential digit access by iterating over recursive depths.
    """
    results = []
    phi_approx = (1 + math.sqrt(5)) / 2  # Golden Ratio
    
    for i in range(1, n+1):
        # Generate digits using recursive fractal transformations
        value = (phi_approx * (2 ** (depth - (i % depth)))) % 1  # Fractalized transformation
        results.append(int(value * 10))  # Extract single digit

    return results
