import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Define the structured chunk data for analysis
chunks_data = {
    "Chunk": list(range(36)),
    "x0": [3]*36,
    "Σx": [14,16,15,10,16,13,16,12,15,16,15,15,11,13,11,9,13,18,11,9,
           15,15,15,14,15,14,15,13,11,9,16,18,15,16,16,14],
    "Top Sequence": [
        [3, 3, 3, 5], [3, 3, 3, 7], [3, 3, 3, 6], [3, 3, 3, 1], [3, 3, 3, 7], [3, 3, 3, 4],
        [3, 3, 3, 7], [3, 3, 3, 3], [3, 3, 3, 6], [3, 4, 3, 6], [3, 3, 3, 6], [3, 4, 3, 5],
        [3, 3, 3, 2], [3, 4, 3, 3], [3, 3, 3, 2], [3, 3, 3, 0], [3, 3, 3, 4], [3, 3, 3, 9],
        [3, 3, 3, 2], [3, 3, 3, 0], [3, 3, 3, 6], [3, 4, 3, 5], [3, 3, 3, 6], [3, 3, 3, 5],
        [3, 3, 3, 6], [3, 3, 3, 5], [3, 3, 3, 6], [3, 3, 3, 4], [3, 3, 3, 2], [3, 3, 3, 0],
        [3, 3, 3, 7], [3, 3, 3, 9], [3, 3, 3, 6], [3, 4, 3, 6], [3, 3, 3, 7], [3, 3, 3, 5]
    ]
}

df_chunks = pd.DataFrame(chunks_data)

# Add a column to classify if the top sequence is a pure [3, 3, 3, x]
df_chunks["is_333x"] = df_chunks["Top Sequence"].apply(lambda x: x[:3] == [3, 3, 3])

# Count how many deviate
non_333x = df_chunks[~df_chunks["is_333x"]]

# Plot Σx vs chunk index to see entropy variation
plt.figure(figsize=(12, 6))
sns.lineplot(data=df_chunks, x="Chunk", y="Σx", label="Σx per Chunk", marker="o")
sns.scatterplot(data=non_333x, x="Chunk", y="Σx", color="red", label="Non-333x Chunks", s=100)
plt.title("Σx Entropy Across Chunks with Deviations Marked")
plt.xlabel("Chunk Index")
plt.ylabel("Σx (Sum of Top Sequence)")
plt.grid(True)
plt.legend()
plt.tight_layout()
plt.show()
