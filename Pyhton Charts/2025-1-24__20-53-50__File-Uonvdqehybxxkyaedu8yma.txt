small_grid_size = 128  # Further reduced size for lattice computation
lattice1_amplitude_small = np.random.random((small_grid_size, small_grid_size))  # Lattice 1 smaller
lattice2_amplitude_small = np.random.random((small_grid_size, small_grid_size))  # Lattice 2 smaller

# Compute differences for the smaller grid
amplitude_diff_small = lattice1_amplitude_small - lattice2_amplitude_small
phase_diff_small = np.angle(lattice1_amplitude_small + 1j * lattice2_amplitude_small) - np.angle(
    lattice2_amplitude_small + 1j * lattice1_amplitude_small
)

# Backpropagated differences for smaller grid
backprop_amplitude_diff_small = np.abs(np.fft.ifft2(np.fft.fft2(amplitude_diff_small)))
backprop_phase_diff_small = np.abs(np.fft.ifft2(np.fft.fft2(phase_diff_small)))
