import numpy as np
import matplotlib.pyplot as plt

# Refined Harmonic Feedback Formula parameters
alpha = 1.5
target = 0.5
iterations = 1000

# Initialize sequence
H = [target]  # Starting with H(0) = target

# Recursive calculation using the formula
for n in range(1, iterations + 1):
    previous_H = H[-1]
    correction = alpha * (target - previous_H) / (n + 1)
    next_H = previous_H * (-0.5) * np.cos(n / np.pi) + correction
    H.append(next_H)

# Analyze convergence
convergence_error = [abs(target - h) for h in H]

# Plot results
plt.figure(figsize=(12, 6))

# Sequence plot
plt.subplot(2, 1, 1)
plt.plot(H, label="H(n)")
plt.axhline(y=target, color='r', linestyle='--', label="Target (0.5)")
plt.title("Convergence of H(n)")
plt.xlabel("Iterations (n)")
plt.ylabel("H(n)")
plt.legend()

# Convergence error plot
plt.subplot(2, 1, 2)
plt.plot(convergence_error, label="|Target - H(n)|")
plt.yscale('log')  # Log scale for better visualization of convergence
plt.title("Convergence Error")
plt.xlabel("Iterations (n)")
plt.ylabel("Error (log scale)")
plt.legend()

plt.tight_layout()
plt.show()
