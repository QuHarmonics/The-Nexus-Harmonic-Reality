# Extract relevant data
atomic_numbers_ext = df_periodic_extended["Atomic Number"]

mass_differences_ext = np.diff(atomic_masses_ext)  # Rate of change in atomic mass

# Compute binary lengths of atomic numbers
binary_lengths_ext = df_periodic_extended["Atomic Number"].apply(lambda x: len(bin(x)[2:]))

# Plot 1: Atomic Number vs. Atomic Mass
