import numpy as np
import matplotlib.pyplot as plt

# Load Zeta zeros (real parts only for simplicity) and simulate some Pi digits (expand later as needed)
# Assuming real Zeta zeros
zeta_zeros = np.array([14.134725, 21.022040, 25.010858, 30.424876, 32.935062, 37.586178, 40.918719])
# Simulated π digits, representing first 20 steps for now
pi_digits = np.array([3, 1, 4, 1, 5, 9, 2, 6, 5, 3, 5, 8, 9, 7, 9, 3, 2, 3, 8, 4])

# Extend Pi digits to create "unfolding" structure - linear progression of decimal values
pi_unfolded = np.cumsum(pi_digits)

# Normalize Zeta zeros and π values to align scales (optional, for visualization purposes)
zeta_normalized = zeta_zeros / np.max(zeta_zeros) * len(pi_unfolded)
pi_normalized = pi_unfolded / np.max(pi_unfolded) * len(pi_unfolded)

# Visualization
plt.figure(figsize=(12, 6))
plt.plot(pi_normalized, label="Unfolded π", linestyle='-', color='blue', marker='o', alpha=0.7)
plt.scatter(zeta_normalized, np.zeros_like(zeta_normalized), label="Zeta Zeros (real)", color='red', s=50, alpha=0.7)
plt.axhline(0, color='black', linestyle='--', linewidth=0.8, alpha=0.5)
plt.xlabel("Index")
plt.ylabel("Normalized Values")
plt.title("Unfolded π vs Zeta Zeros on a Line")
plt.legend()
plt.grid(alpha=0.3)
