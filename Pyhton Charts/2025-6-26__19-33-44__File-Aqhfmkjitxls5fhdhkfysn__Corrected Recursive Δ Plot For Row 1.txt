Corrected Recursive Δ Plot for Row 1

# Redefine Row 1 from Pi digits or simulated values
row = [1, 6, 7, 7, 7, 2, 1, 6]

# Initialize cumulative counter using pairwise delta between current and previous values
# but starting from the first value
cumulative = [row[0]]  # Start from the first element

for i in range(1, len(row)):
    delta = row[i] - row[i - 1]
    new_value = cumulative[-1] + delta
    cumulative.append(new_value)

# Plotting the corrected cumulative delta transformation
x_vals = [i + 1 for i in range(len(row))]

plt.figure(figsize=(10, 6))
plt.plot(x_vals, cumulative, marker='o', linestyle='-', linewidth=2, color='darkgreen', label="Row 1 (Corrected Δ Path)")
plt.title("Corrected Recursive Δ Plot for Row 1")
plt.xlabel("Time Step")
plt.ylabel("Running Counter (Δ Applied)")
plt.grid(True)
plt.legend()
plt.tight_layout()
plt.show()
