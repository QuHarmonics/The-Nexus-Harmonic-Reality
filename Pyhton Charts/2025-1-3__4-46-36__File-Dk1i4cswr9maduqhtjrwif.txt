import numpy as np
import matplotlib.pyplot as plt

# Constants for both simulations
A = 0.35  # Harmonic damping constant
H = 0.5   # Critical equilibrium (midpoint)
Delta = 0.01  # Imbalance from the pull of time (1% above and below equilibrium)
time = np.linspace(0, 10, 1000)  # Time steps

# Growth rates
growth_rate_fixed = 1.5  # Fixed growth rate for the first code
phi = (1 + np.sqrt(5)) / 2  # Golden ratio for the second code

# Define the harmonic wave equation
def harmonic_wave(t, A, H, Delta, growth_rate):
    return H + A * np.sin(growth_rate * t) * np.exp(-0.35 * t) + Delta * np.sign(np.sin(growth_rate * t))

# Calculate waveforms for fixed growth rate (1.5)
waveform_fixed_upper = harmonic_wave(time, A, H, Delta, growth_rate_fixed)
waveform_fixed_lower = harmonic_wave(time, A, H, -Delta, growth_rate_fixed)

# Calculate waveforms for golden ratio growth rate (phi)
waveform_phi_upper = harmonic_wave(time, A, H, Delta, phi)
waveform_phi_lower = harmonic_wave(time, A, H, -Delta, phi)

# Plot the waveforms
plt.figure(figsize=(14, 8))

# Plot for fixed growth rate
plt.plot(time, waveform_fixed_upper, label='Fixed Growth (Upper, +Delta)', color='blue', alpha=0.7)
plt.plot(time, waveform_fixed_lower, label='Fixed Growth (Lower, -Delta)', color='red', alpha=0.7)

# Plot for golden ratio growth rate
plt.plot(time, waveform_phi_upper, label='Golden Ratio Growth (Upper, +Delta)', color='green', linestyle='--')
plt.plot(time, waveform_phi_lower, label='Golden Ratio Growth (Lower, -Delta)', color='orange', linestyle='--')

# Add equilibrium line
plt.axhline(H, color='black', linestyle='-.', label='Equilibrium (H = 0.5)')

# Titles and labels
plt.title("Comparison of Harmonic Waves: Fixed Growth vs Golden Ratio Growth")
plt.xlabel("Time")
plt.ylabel("Amplitude")
plt.legend()
plt.grid()

# Display the plot
plt.show()
