# Plot the triangles sequentially along the x-axis
x_offset = 0
x_spacing = 5  # Space between each triangle

    # Calculate C using geometric constraints

        Cy = np.sqrt(abs(a**2 - (Cx - x_offset)**2))  # Height
    else:
        Cx, Cy = x_offset, 0

    C = (Cx, Cy)

    x_coords = [A[0], B[0], C[0], A[0]]
    y_coords = [A[1], B[1], C[1], A[1]]
    plt.plot(x_coords, y_coords, marker='o', linestyle='-', label=f"Triangle {i+1}")
