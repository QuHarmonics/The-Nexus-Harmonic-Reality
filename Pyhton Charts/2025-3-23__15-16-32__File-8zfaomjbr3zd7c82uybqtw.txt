import matplotlib.pyplot as plt
import numpy as np

# Simulate Byte1 construction harmonically as waveform loops
# Steps from Byte1 sequence: (1, 4, 1, 5, 9, 2, 6, 5)

byte1_sequence = [1, 4, 1, 5, 9, 2, 6, 5]

# Map each value to a sine wave of increasing frequency
x = np.linspace(0, 2 * np.pi, 1000)
plt.figure(figsize=(12, 8))

for idx, val in enumerate(byte1_sequence):
    freq = val
    phase_shift = idx * 0.3  # offset to create a visual spiral effect
    y = np.sin(freq * x + phase_shift) / (idx + 1)  # diminishing amplitude
    plt.plot(x, y, label=f'Step {idx+1}: {val}')

plt.title('Harmonic Mapping of Byte1 Recursive Sequence')
plt.xlabel('Time (x)')
plt.ylabel('Amplitude')
plt.legend()
plt.grid(True)
plt.show()
