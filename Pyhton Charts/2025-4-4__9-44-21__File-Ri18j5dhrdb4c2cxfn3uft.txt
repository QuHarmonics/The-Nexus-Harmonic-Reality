# Re-define the radial plot function (it got lost in the earlier cell)
import matplotlib.pyplot as plt
import numpy as np

def radial_plot(data, title, color):
    n = len(data)
    angles = np.linspace(0, 2 * np.pi, n, endpoint=False)
    radii = data
    x = radii * np.cos(angles)
    y = radii * np.sin(angles)

    plt.figure(figsize=(6, 6))
    plt.plot(x, y, color=color)
    plt.fill(x, y, color=color, alpha=0.3)
    plt.title(title)
    plt.axis('equal')
    plt.axis('off')
    plt.tight_layout()
    plt.show()
