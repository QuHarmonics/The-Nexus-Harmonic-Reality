import numpy as np
import matplotlib.pyplot as plt
import math

# Define range
n_values = np.arange(1, 1025)

# Calculate distance to next power of 2 and the ratio
next_power_of_2 = np.power(2, np.ceil(np.log2(n_values)))
distances = next_power_of_2 - n_values
ratios = distances / n_values

# Plotting
plt.figure(figsize=(14, 7))
plt.plot(n_values, ratios, label='Distance to Next Power of 2 / n', color='blue')

# Plot horizontal reference lines for pi and golden ratio
plt.axhline(y=math.pi, color='red', linestyle='--', label='π ≈ 3.14159')
plt.axhline(y=(1 + math.sqrt(5)) / 2, color='green', linestyle='--', label='Golden Ratio ≈ 1.618')

plt.title("Binary Distance Ratio to Next Power of 2")
plt.xlabel("n")
plt.ylabel("Distance to 2^ceil(log2(n)) / n")
plt.grid(True)
plt.legend()
plt.tight_layout()
plt.show()
