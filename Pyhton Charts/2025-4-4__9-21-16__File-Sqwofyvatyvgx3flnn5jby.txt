import matplotlib.pyplot as plt
import numpy as np
import hashlib

# === Helper Functions ===

def hex_string_to_bytes(hex_str):
    hex_str = hex_str.replace(' ', '').replace('\n', '')
    return bytes.fromhex(hex_str)

def hex_to_rgb_array(hex_str):
    hex_str = hex_str.replace('\n', '').replace(' ', '')
    bytes_array = [hex_str[i:i+2] for i in range(0, len(hex_str), 2)]
    rgb_array = []
    for i in range(0, len(bytes_array), 3):
        if i+2 < len(bytes_array):
            r = int(bytes_array[i], 16)
            g = int(bytes_array[i+1], 16)
            b = int(bytes_array[i+2], 16)
            rgb_array.append((r, g, b))
    return rgb_array

def byte_to_bit_array(byte_val):
    return [(byte_val >> i) & 1 for i in reversed(range(8))]

def hex_to_binary_grid(hex_str):
    byte_vals = [int(hex_str[i:i+2], 16) for i in range(0, len(hex_str), 2)]
    bit_grid = np.array([byte_to_bit_array(b) for b in byte_vals])
    return bit_grid
