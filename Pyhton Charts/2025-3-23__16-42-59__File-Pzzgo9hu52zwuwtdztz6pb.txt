# Calculate phase zones: segment the 256 digits into 8 segments (32 digits each)
segment_size = 32
num_segments = 256 // segment_size

odd_segments = []
even_segments = []

for i in range(num_segments):
    segment = pi_digits_256[i * segment_size: (i + 1) * segment_size]
    odd_count = sum(1 for d in segment if d % 2 != 0)
    even_count = segment_size - odd_count
    odd_segments.append(odd_count)
    even_segments.append(even_count)

# Plot segment-wise odd vs even counts
labels = [f"Zone {i+1}" for i in range(num_segments)]
x = range(num_segments)

plt.figure(figsize=(14, 6))
plt.bar(x, odd_segments, width=0.4, label="Odd Digits", align='center')
plt.bar(x, even_segments, width=0.4, bottom=odd_segments, label="Even Digits", align='center')
plt.xticks(x, labels)
plt.ylabel("Digit Count per Segment")
plt.title("Odd/Even Distribution by Harmonic Phase Zones (Pi's First 256 Digits)")
plt.legend()
plt.grid(True)
