import numpy as np
import matplotlib.pyplot as plt

# Convert the digit stream into RGB logic maps
digit_chars = [
    int(c) for c in (
        "3330333433333337363136323631333933363339333433303631333333363339333236313337333833373330333836313331"
        "3636363233353634333736323632333036333335363633393339333236323332333336313331363636323333363236343334"
        "333833323636363433313634333436323633363433343636343334"
    ) if c.isdigit()
]

# Create 3-channel logic maps by treating digit values as intensity per channel
length = len(digit_chars)
r = np.array([(v % 3 == 0) * v / 9.0 for v in digit_chars])
g = np.array([(v % 3 == 1) * v / 9.0 for v in digit_chars])
b = np.array([(v % 3 == 2) * v / 9.0 for v in digit_chars])

# Stack into a (length x 3) RGB array for visualization
rgb_matrix = np.stack([r, g, b], axis=1).reshape(1, length, 3)

# Plot the RGB echo field
plt.figure(figsize=(20, 2))
plt.imshow(rgb_matrix, aspect='auto')
plt.title("Symbolic SHA Echo RGB Overlay Field")
plt.axis('off')
plt.tight_layout()
plt.show()
